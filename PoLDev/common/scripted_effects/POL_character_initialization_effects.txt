	limit = { 
		NOR = { # Can't change faith of holy orders or Heads of Faith
			any_held_title = {
				is_holy_order = yes
			}
			AND = {
				exists = faith.religious_head
				faith.religious_head = this
			}
		}
	}
	
	set_character_faith = faith:dragonblooded_silence_religion
	
}
